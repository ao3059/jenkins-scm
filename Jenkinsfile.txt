pipeline {
    agent any
    stages {
        stage('Checkout Code') {
            steps {
                checkout scm
            }
        }
        stage('Install Node.js and npm') {
            steps {
                sh 'nvm install 18'
                sh 'nvm use 18'
                sh 'node -v'
                sh 'npm -v'
            }
        }
        stage('Install Dependencies') {
            steps {
                sh 'npm install'
            }
        }
        stage('Run Tests') {
            steps {
                sh 'npm test'
            }
        }
        stage('Build Docker Image') {
            steps {
                sh 'docker build -t my-ecommerce-app .'
            }
        }
        stage('Run Docker Container') {
            steps {
                sh 'docker run -d -p 8081:80 my-ecommerce-app'
            }
        }
    }
    post {
        always {
            echo 'This will always run, whether the pipeline succeeds or fails.'
        }
        failure {
            echo 'Pipeline failed!'
        }
    }
}
